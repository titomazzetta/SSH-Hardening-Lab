# Example UFW rules for the hardened SSH server.
# These commands are intended to be run manually with sudo.

sudo ufw default deny incoming
sudo ufw default allow outgoing

# Allow SSH from anywhere (for lab). For production, restrict this.
sudo ufw allow OpenSSH
